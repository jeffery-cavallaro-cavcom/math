\section*{Cayley Digraphs}

In a Caley digraph for a group $G$ generated by a subset $S$:
\begin{itemize}
\item Each element in $G$ is represented by a node.
\item Each node has an outgoing arc for each generator in $S$.
\item Each node has an incoming arc for each generator in $S$.
\item Generators that are their own inverse are represented by an undirected
  arc.
\end{itemize}

\begin{example}
  \begin{minipage}[t]{3.5in}
    $G=Z_6$ with $S={2,3}$

    \bigskip

    \begin{tikzpicture}
      \node [circle,draw] (0) at (0,6) {$0$};
      \node [circle,draw] (1) at (4,0) {$1$};
      \node [circle,draw] (2) at (0,3) {$2$};
      \node [circle,draw] (3) at (4,6) {$3$};
      \node [circle,draw] (4) at (0,0) {$4$};
      \node [circle,draw] (5) at (4,3) {$5$};
      \draw (0) edge [-{>[scale=1.5]}] node [fill=white] {$2$} (2);
      \draw (2) edge [-{>[scale=1.5]}] node [fill=white] {$2$} (4);
      \draw (4) edge [bend left,-{>[scale=1.5]}] node [fill=white] {$2$} (0);
      \draw (3) edge [-{>[scale=1.5]}] node [fill=white] {$2$} (5);
      \draw (5) edge [-{>[scale=1.5]}] node [fill=white] {$2$} (1);
      \draw (0) edge node [fill=white] {$3$} (3);
      \draw (2) edge node [fill=white] {$3$} (5);
      \draw (4) edge node [fill=white] {$3$} (1);
      \draw (1) edge [bend right,-{>[scale=1.5]}] node [fill=white] {$2$} (3);
    \end{tikzpicture}
  \end{minipage}
  \begin{minipage}[t]{3in}
    $G=K_4$ with $S={a,b}$

    \bigskip

    \begin{tikzpicture}
      \node [circle,draw] (e) at (0,4) {$e$};
      \node [circle,draw] (a) at (4,4) {$a$};
      \node [circle,draw] (b) at (4,0) {$b$};
      \node [circle,draw] (c) at (0,0) {$c$};
      \draw (e) edge node [fill=white] {$a$} (a);
      \draw (b) edge node [fill=white] {$a$} (c);
      \draw (e) edge node [fill=white] {$b$} (b);
      \draw (a) edge node [fill=white] {$b$} (c);
    \end{tikzpicture}
  \end{minipage}
\end{example}
